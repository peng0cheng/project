【增加命名空间】

namespace 空间名
【增加个命名空间】

namespace 空间名{属于该空间内的代码}
【增加多层命名空间】

namespace 空间名\子空间名...
【调用当前空间的函数】

函数名( )
【调用当前空间的常量】

常量名
【调用当前空间的类】

类名
【调用全局空间的函数】

\ 函数名( )
【调用全局空间的常量】

\ 常量名
【调用全局空间的类】

\ 类名
【调用指定空间的函数】

\空间名\函数名( )
【调用指定空间的常量】

\空间名\常量名
【调用指定空间的类】

\空间名\类名
【用魔术常量调用当前空间的内容】

__NAMESPACE__.\'内容'
【用关键字调用当前空间的内容】

namespace\内容
【命名空间起名】

use namespace 空间名 as 新空间名
【命名空间起名】

use namespace 空间名
【别名调用当前空间】

新空间名\内容
【导入全局空间的类】

use\类名
【导入指定空间的类】

use\空间名\类名
